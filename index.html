Harjutus 28.
Valgusfoor – (foori moodul on olemas) kui auto hakkab foorile lähenema, siis kollane vilgub ja süttib punane. Pärast ootamist näidatakse rohelist. Kuva info ka Serial Monitoris.

Kood:

#define trigPin 3
#define echoPin 2
long duration;
int distance;

void setup()
{
pinMode(trigPin, OUTPUT);
pinMode(echoPin, INPUT);
pinMode(4, OUTPUT);
pinMode(5, OUTPUT);
pinMode(6, OUTPUT);
Serial.begin(9600);
Serial.println("1- ON(1) OFF(4)");
Serial.println("2- ON(2) OFF(5)");
Serial.println("3- ON(3) OFF(6)");
Serial.println("Check Distance (7)");
digitalWrite(4, HIGH);
}
void loop() {
digitalWrite(trigPin, LOW);
delayMicroseconds(2);
digitalWrite(trigPin, HIGH);
delayMicroseconds(10);
digitalWrite(trigPin, LOW);
duration = pulseIn(echoPin, HIGH);
distance = duration * 0.034 / 2;

if (Serial.available()){int state = Serial.parseInt();
if (state == 1){digitalWrite(4, HIGH);
Serial.println("Red LED turned ON");
}
if (state == 2){digitalWrite(5, HIGH);
Serial.println("Yellow LED turned ON");
}
if (state == 3){digitalWrite(6, HIGH);
Serial.println("Green LED turned ON");
}
if (state == 4){digitalWrite(4, LOW);
Serial.println("Red LED turned OFF");
}
if (state == 5){digitalWrite(5, LOW);
Serial.println("Yellow LED turned OFF");
}
if (state == 6){digitalWrite(6, LOW);
Serial.println("Green LED turned OFF") ;
}
if (state == 7){
Serial.print("Distance: ");
Serial.print(distance);
Serial.println(" cm");
}
}
if (distance < 20){
delay(1000);
digitalWrite(4, LOW);
delay(100);
digitalWrite(5, HIGH);
delay(1000);
digitalWrite(5, LOW);
delay(1000);
digitalWrite(5, HIGH);
delay(1000);
digitalWrite(5, LOW);
delay(200);
digitalWrite(6, HIGH);
delay(10000);
digitalWrite(6, LOW);
digitalWrite(4, HIGH);
}
}
